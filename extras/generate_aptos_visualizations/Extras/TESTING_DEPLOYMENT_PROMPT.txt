def _get_augmentations(self, image_batch):
    """16 deterministic augmentations for maximum variance reduction"""
    
    # Group A: Geometric (8 views)
    h_flip = torch.flip(image_batch, dims=[3])
    v_flip = torch.flip(image_batch, dims=[2])
    hv_flip = torch.flip(torch.flip(image_batch, dims=[3]), dims=[2])
    rot90 = torch.rot90(image_batch, k=1, dims=[2, 3])
    rot90_h = torch.flip(rot90, dims=[3])
    rot180 = torch.rot90(image_batch, k=2, dims=[2, 3])
    rot270 = torch.rot90(image_batch, k=3, dims=[2, 3])
    
    # Group B: Photometric (4 views)
    bright_up = torch.clamp(image_batch * 1.10, min=-3.0, max=3.0)
    bright_down = torch.clamp(image_batch * 0.90, min=-3.0, max=3.0)
    contrast_up = torch.clamp(image_batch * 1.10, min=-3.0, max=3.0)
    contrast_down = torch.clamp(image_batch * 0.90, min=-3.0, max=3.0)
    
    # Group C: Color channel shifts (4 views)
    # Shift individual channels slightly (after normalization)
    red_shift = image_batch.clone()
    red_shift[:, 0, :, :] = torch.clamp(red_shift[:, 0, :, :] * 1.05, min=-3.0, max=3.0)
    
    green_shift = image_batch.clone()
    green_shift[:, 1, :, :] = torch.clamp(green_shift[:, 1, :, :] * 1.05, min=-3.0, max=3.0)
    
    blue_shift = image_batch.clone()
    blue_shift[:, 2, :, :] = torch.clamp(blue_shift[:, 2, :, :] * 1.05, min=-3.0, max=3.0)
    
    balanced = torch.clamp(image_batch * 1.02, min=-3.0, max=3.0)
    
    return [
        image_batch,      # 1. Original
        h_flip,          # 2. H-flip
        v_flip,          # 3. V-flip
        hv_flip,         # 4. H+V flip
        rot90,           # 5. Rotate 90
        rot90_h,         # 6. Rotate 90 + H-flip
        rot180,          # 7. Rotate 180
        rot270,          # 8. Rotate 270
        bright_up,       # 9. Brightness +10%
        bright_down,     # 10. Brightness -10%
        contrast_up,     # 11. Contrast +10%
        contrast_down,   # 12. Contrast -10%
        red_shift,       # 13. Red channel +5%
        green_shift,     # 14. Green channel +5%
        blue_shift,      # 15. Blue channel +5%
        balanced,        # 16. Balanced boost
    ]
